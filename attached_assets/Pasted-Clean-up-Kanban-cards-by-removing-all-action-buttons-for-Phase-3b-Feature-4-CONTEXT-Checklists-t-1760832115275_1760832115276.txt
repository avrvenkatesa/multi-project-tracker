Part 2: Keep Only Essential Card Information

The Kanban card should ONLY contain:
```html
<div class="kanban-card" onclick="openIssueDetail(${item.id})">
  <!-- Checkbox for bulk select (keep this) -->
  <input 
    type="checkbox" 
    class="issue-checkbox" 
    data-issue-id="${item.id}"
    onclick="event.stopPropagation()"
  />
  
  <!-- Card header -->
  <div class="card-header">
    <span class="item-type">${item.type || 'issue'}</span>
    <span class="priority-badge">${item.priority}</span>
  </div>
  
  <!-- Title (keep clickable to open detail) -->
  <h3 class="card-title">${item.title}</h3>
  
  <!-- Description preview (if exists) -->
  ${item.description ? `
    <p class="card-description text-sm text-gray-600">
      ${item.description.substring(0, 100)}...
    </p>
  ` : ''}
  
  <!-- Assignee (keep) -->
  <div class="card-assignee">
    <span class="assignee-icon">👤</span>
    <span>${item.assignee || 'Unassigned'}</span>
  </div>
  
  <!-- Due date (keep) -->
  ${item.due_date ? `
    <div class="card-due-date ${isOverdue(item.due_date) ? 'overdue' : ''}">
      ${formatDueDate(item.due_date)}
    </div>
  ` : ''}
  
  <!-- Tags (keep) -->
  ${item.tags && item.tags.length > 0 ? `
    <div class="card-tags">
      ${item.tags.map(tag => `
        <span class="tag">${tag}</span>
      `).join('')}
    </div>
  ` : ''}
</div>
```
